    94→  const createSaleMutation = useMutation({
    95→    mutationFn: async ({
    96→      paymentMethod,
    97→      amountPaid,
    98→      customerName,
    99→      customerPhone,
   100→    }: {
   101→      paymentMethod: PaymentMethod;
   102→      amountPaid: number;
   103→      customerName?: string;
   104→      customerPhone?: string;
   105→    }) => {
   106→      if (Date.now() - lastSubmitRef.current < 2000) {
   107→        throw new Error("Vente déjà en cours de traitement");
   108→      }
   109→      lastSubmitRef.current = Date.now();
   110→      // Obtenir le numéro de vente (fallback vers timestamp+uuid si RPC non disponible)
   111→      let finalSaleNumber = '';
   112→      try {
   113→        const { data: saleNumber, error: rpcError } = await supabase.rpc("generate_sale_number");
   114→        if (rpcError) {
   115→          // generate_sale_number RPC indisponible, utilisation du fallback
   116→        }
   117→        finalSaleNumber = saleNumber || '';
   118→      } catch {
   119→        // generate_sale_number RPC indisponible, utilisation du fallback
   120→      }
   121→      // Fallback : utiliser un fragment crypto.randomUUID — pas de risque de collision entre terminaux
   122→      if (!finalSaleNumber) {
   123→        const uid = crypto.randomUUID().replace(/-/g, '').substring(0, 12).toUpperCase();
   124→        finalSaleNumber = `VTE-${uid}`;
   125→      }
   126→
   127→      const subtotal = cart.reduce(
   128→        (sum, item) => sum + item.product.price * item.quantity,
   129→        0
   130→      );
   131→      const taxAmount = cart.reduce((sum, item) => {
   132→        const t = computeTax(item.product.price, item.product.tax_rate, orgTaxRate);
   133→        return sum + t.taxAmount * item.quantity;
   134→      }, 0);
   135→      // Les prix sont TTC (taxe incluse) : le sous-total inclut déjà la taxe
   136→      // totalMontant = sous-total (ce que le client paie)
   137→      // Pour la DB : stocker le sous-total comme total TTC, montantTaxe comme la part de taxe
   138→      const totalAmount = subtotal;
   139→      const htAmount = subtotal - taxAmount;
   140→      const changeAmount = amountPaid - totalAmount;
   141→
   142→      // Essayer d'abord la création atomique de vente via RPC (C4 : transaction unique)
   143→      const saleItems = cart.map((item) => ({
   144→        product_id: item.product.id,
   145→        product_name: item.product.name,
   146→        quantity: item.quantity,
   147→        unit_price: item.product.price,
   148→        total_price: item.product.price * item.quantity,
   149→      }));
   150→
   151→      const orgId = profile?.organization_id || null;
   152→      let creditUpdateFailed = false;
   153→
   154→      // ⚠️ create_full_sale RPC est OBLIGATOIRE — plus de fallback non-atomique.
   155→      // L'ancien chemin (INSERT sale + INSERT items + batch_update_stock) était
   156→      // non-atomique : une panne entre les étapes laissait des données incohérentes.
   157→      const { data: rpcSaleId, error: rpcError } = await supabase.rpc("create_full_sale", {
   158→        p_user_id: user!.id,
   159→        p_organization_id: orgId,
   160→        p_sale_number: finalSaleNumber,
   161→        p_subtotal: htAmount,
   162→        p_tax_amount: taxAmount,
   163→        p_total_amount: totalAmount,
   164→        p_payment_method: paymentMethod,
   165→        p_amount_paid: amountPaid,
   166→        p_change_amount: changeAmount > 0 ? changeAmount : 0,
   167→        p_customer_name: customerName || null,
   168→        p_customer_phone: customerPhone || null,
   169→        p_seller_name: profile?.owner_name || null,
   170→        p_items: saleItems,
   171→      });
   172→
   173→      if (rpcError || !rpcSaleId) {
   174→        throw new Error(
   175→          `Impossible de créer la vente (RPC create_full_sale) : ${rpcError?.message || 'Réponse vide'}. Veuillez réessayer.`
   176→        );
   177→      }
   178→
   179→      // Récupérer l'enregistrement de vente pour le ticket
   180→      const { data: sale } = await supabase
   181→        .from("sales")
   182→        .select("id, sale_number, payment_method, amount_paid, customer_name, customer_phone")
   183→        .eq("id", rpcSaleId)
   184→        .single();
   185→
   186→      if (!sale) {
   187→        throw new Error('Vente créée mais introuvable. Veuillez réessayer.');
   188→      }
   189→
   190→      // Si vente à crédit, créer une entrée customer_credits pour le suivi de la dette
   191→      if (paymentMethod === "credit" && totalAmount > 0) {
   192→        let customerId: string | null = null;
   193→
   194→        // Upsert atomique du client — évite la race condition SELECT→INSERT
   195→        // où deux vendeurs concurrents créeraient des doublons pour le même téléphone.
   196→        // La contrainte unique sur customers(phone, organization_id) garantit l'atomicité.
   197→        if (customerPhone) {
   198→          const upsertData: Record<string, unknown> = {
   199→            name: customerName || customerPhone,
   200→            phone: customerPhone,
   201→          };
   202→          if (profile?.organization_id) {
   203→            upsertData.organization_id = profile.organization_id;
   204→          }
   205→          const { data: upsertedCustomer, error: custErr } = await supabase
   206→            .from("customers")
   207→            .upsert(upsertData as never, {
   208→              onConflict: 'phone,organization_id',
   209→              ignoreDuplicates: false,
   210→            })
   211→            .select("id")
   212→            .maybeSingle();
   213→
   214→          if (!custErr && upsertedCustomer) {
   215→            customerId = upsertedCustomer.id;
   216→          }
   217→        } else if (customerName) {
   218→          // Pas de téléphone mais nom fourni — essayer de trouver par nom dans l'organisation
   219→          const { data: existingCustomer } = await supabase
   220→            .from("customers")
   221→            .select("id")
   222→            .eq("name", customerName)
   223→            .eq("organization_id", orgId)
   224→            .maybeSingle();
   225→          if (existingCustomer) {
   226→            customerId = existingCustomer.id;
   227→          }
   228→        }
   229→
   230→        if (customerId) {
   231→          const creditInsert: Record<string, unknown> = {
   232→            user_id: user!.id,
   233→            customer_id: customerId,
   234→            sale_id: sale.id,
   235→            amount: totalAmount,
   236→            type: "credit",
   237→            description: `Vente crédit ${finalSaleNumber}`,
   238→          };
   239→          if (profile?.organization_id) {
   240→            creditInsert.organization_id = profile.organization_id;
   241→          }
   242→          await supabase.from("customer_credits").insert(creditInsert as never);
   243→
   244→          // Mettre à jour total_credit du client atomiquement via RPC
   245→          const { error: creditUpdateError } = await supabase.rpc("increment_customer_credit", {
   246→            p_customer_id: customerId,
   247→            p_amount: totalAmount,
   248→          });
   249→
   250→          if (creditUpdateError) {
   251→            // La vente est déjà enregistrée mais la mise à jour du crédit a échoué.
   252→            // On ne fait PAS de fallback non-atomique pour éviter les incohérences.
   253→            // L'utilisateur est notifié pour pouvoir relancer la synchronisation.
   254→            reportError(new Error(`increment_customer_credit RPC failed: ${creditUpdateError.message}`));
   255→            // Stocker l'erreur pour notification dans onError
   256→            creditUpdateFailed = true;
   257→          }
   258→        }
   259→      }
   260→
   261→      return { sale, changeAmount, creditUpdateFailed };
   262→    },
   263→    onSuccess: ({ sale, changeAmount, creditUpdateFailed }) => {
   264→      // Calculer la taxe pour l'affichage du ticket AVANT clearCart()
   265→      const receiptTaxAmount = cart.reduce((sum, item) => {
   266→        const t = computeTax(item.product.price, item.product.tax_rate, orgTaxRate);
   267→        return sum + t.taxAmount * item.quantity;
   268→      }, 0);
   269→      const receiptSubtotal = cartTotal - receiptTaxAmount;
   270→
   271→      // Préparer les données du ticket pour le dialogue (avant clearCart)
   272→      const receiptData: ReceiptData = {
   273→        saleNumber: sale.sale_number,
   274→        date: new Date(),
   275→        items: cart.map((item) => ({
   276→          product_name: item.product.name,
   277→          quantity: item.quantity,
   278→          unit_price: item.product.price,
   279→          total_price: item.product.price * item.quantity,
   280→        })),
   281→        subtotal: receiptSubtotal,
   282→        total: cartTotal,
   283→        paymentMethod: sale.payment_method,
   284→        amountPaid: sale.amount_paid,
   285→        change: changeAmount > 0 ? changeAmount : 0,
   286→        customerName: sale.customer_name || undefined,
   287→        customerPhone: sale.customer_phone || undefined,
   288→        businessName: profile?.business_name || "Ma Boutique",
   289→        businessAddress: profile?.address || undefined,
   290→        businessPhone: profile?.phone || undefined,
   291→        sellerName: profile?.owner_name || undefined,
   292→        currencySymbol: currency.displaySymbol || currency.symbol,
   293→        currencyPosition: currency.position,
   294→        logoUrl: settings?.logo_url || branding.logoUrl,
   295→        template: branding.receiptTemplate,
   296→        paperSize: (settings?.extra_settings as Record<string, string>)?.receiptPaperSize as ReceiptData["paperSize"] || "80mm",
   297→        showLogo: settings?.receipt_show_logo ?? true,
   298→        showTax: settings?.receipt_show_tax ?? true,
   299→        footerText: settings?.receipt_footer || undefined,
   300→        organizationId: profile?.organization_id ?? undefined,
   301→        taxRate: orgTaxRate,
   302→      };
   303→
   304→      // Clear cart AFTER receipt data is computed
   305→      clearCart();
   306→      setLastReceiptData(receiptData);
   307→      setIsReceiptOpen(true);
   308→
   309→      queryClient.invalidateQueries({ queryKey: ["products"] });
   310→      queryClient.invalidateQueries({ queryKey: ["sales"] });
   311→      setIsPaymentOpen(false);
   312→
   313→      // Avertir l'utilisateur si la mise à jour du crédit a échoué
   314→      if (creditUpdateFailed) {
   315→        toast({
   316→          variant: "destructive",
   317→          title: "Vente enregistrée, crédit en attente",
   318→          description: "La vente est validée mais la mise à jour du crédit client a échoué. Vérifiez les crédits du client.",
   319→          duration: 8000,
   320→        });
   321→      }
   322→    },
   323→    onError: (error: unknown) => {
   324→      // Les erreurs Supabase sont des objets simples, pas des instances Error
   325→      let message = "Impossible d'enregistrer la vente";
   326→      if (error instanceof Error) {
   327→        message = error.message;
   328→      } else if (typeof error === 'object' && error !== null) {
   329→        const err = error as Record<string, unknown>;
   330→        if (typeof err.message === 'string') {
   331→          message = err.message;
   332→        } else if (typeof err.details === 'string') {
   333→          message = err.details;
   334→        } else if (typeof err.code === 'string') {
   335→          message = `Erreur ${err.code}: veuillez réessayer`;
   336→        } else {
   337→          try {
   338→            message = JSON.stringify(error);
   339→          } catch {
   340→            message = String(error);
   341→          }
   342→        }
   343→      } else {
   344→        message = String(error);
   345→      }
   346→      toast({
   347→        variant: "destructive",
   348→        title: "Erreur de vente",
   349→        description: message,
   350→      });
   351→      reportError(error instanceof Error ? error : new Error(message));
   352→    },
   353→  });
   354→
   355→  const addToCart = useCallback((product: Product, addQty: number = 1) => {
   356→    const existing = cart.find((item) => item.product.id === product.id);
   357→    const currentQty = existing?.quantity || 0;
   358→    const targetQty = currentQty + addQty;
   359→
   360→    if (targetQty > product.stock_quantity) {
   361→      toast({
   362→        variant: "destructive",
   363→        title: "Stock insuffisant",
   364→        description: `Seulement ${product.stock_quantity} ${product.unit || "unité(s)"} disponible(s)`,
   365→      });
   366→      return;
   367→    }
   368→    addToCartStore(product, addQty);
   369→  }, [cart, toast, addToCartStore]);
   370→
   371→  const updateCartQuantity = useCallback((productId: string, quantity: number) => {
   372→    if (quantity <= 0) {
   373→      removeItem(productId);
   374→      return;
   375→    }
   376→    const item = cart.find((i) => i.product.id === productId);
   377→    if (item && quantity > item.product.stock_quantity) {
   378→      toast({
   379→        variant: "destructive",
   380→        title: "Stock insuffisant",
   381→        description: `Seulement ${item.product.stock_quantity} disponible(s)`,
   382→      });
   383→      return;
   384→    }
   385→    updateQuantity(productId, quantity);
   386→  }, [cart, toast, updateQuantity, removeItem]);
   387→
   388→  const removeFromCart = useCallback((productId: string) => {
   389→    removeItem(productId);
   390→  }, [removeItem]);
   391→
   392→  const handleBarcodeScan = async (barcode: string) => {
   393→    // Recherche serveur du produit par code-barres
   394→    try {
   395→      const found = await lookupBarcode(barcode, profile?.organization_id);
   396→      if (found) {
   397→        addToCart(found as Product);
   398→      } else {
   399→        setGridSearchInput(barcode);
   400→        toast({
   401→          variant: "destructive",
   402→          title: "Produit non trouvé",
   403→          description: `Aucun produit avec le code-barres: ${barcode}`,
   404→        });
   405→      }
   406→    } catch {
   407→      setGridSearchInput(barcode);
   408→      toast({
   409→        variant: "destructive",
   410→        title: "Produit non trouvé",
   411→        description: `Aucun produit avec le code-barres: ${barcode}`,
   412→      });
   413→    }
   414→  };
   415→
   416→  // Les filtres (catégorie, stock, recherche) sont gérés côté serveur par usePOSProducts
   417→
   418→  const itemCount = cart.reduce((sum, item) => sum + item.quantity, 0);
   419→
   420→  // Raccourcis clavier
   421→  usePOSKeyboardShortcuts({
   422→    onFocusSearch: useCallback(() => {
   423→      searchInputRef.current?.focus();
   424→    }, []),
   425→    onOpenPayment: useCallback(() => setIsPaymentOpen(true), []),
   426→    onClearCart: useCallback(() => clearCart(), [clearCart]),
   427→    onToggleView: useCallback(() => setViewMode((v) => v === "grid" ? "list" : "grid"), []),
   428→    onToggleOutOfStock: useCallback(() => setShowOutOfStock((v) => !v), []),
   429→    onOpenScanner: useCallback(() => setIsScannerOpen(true), []),
   430→    onShowHelp: useCallback(() => setShowShortcuts(true), []),
   431→    onConfirmPayment: useCallback(() => {
   432→      confirmPaymentRef.current?.();
   433→    }, []),
   434→    onIncrementLastItem: useCallback(() => {
   435→      if (cart.length === 0) return;
   436→      const lastItem = cart[cart.length - 1];
   437→      const newQty = lastItem.quantity + 1;
   438→      if (newQty <= lastItem.product.stock_quantity) {
   439→        updateQuantity(lastItem.product.id, newQty);
   440→      }
   441→    }, [cart, updateQuantity]),
   442→    onDecrementLastItem: useCallback(() => {
   443→      if (cart.length === 0) return;
   444→      const lastItem = cart[cart.length - 1];
   445→      const newQty = lastItem.quantity - 1;
   446→      if (newQty <= 0) {
   447→        removeItem(lastItem.product.id);
   448→      } else {
   449→        updateQuantity(lastItem.product.id, newQty);
   450→      }
   451→    }, [cart, updateQuantity, removeItem]),
   452→    hasCartItems: cart.length > 0,
   453→    isPaymentOpen,
   454→  });
   455→
   456→  return (
   457→    <DashboardLayout>
   458→      <div className="h-[calc(100dvh-10rem)] sm:h-[calc(100dvh-9rem)] md:h-[calc(100dvh-8rem)] lg:h-[calc(100dvh-2.5rem)] flex flex-col lg:flex-row gap-3 lg:gap-4">
   459→        {/* Products Section */}
   460→        <div className="flex-1 flex flex-col min-h-0">
   461→          {/* Search and Categories */}
   462→          <div className="space-y-3 mb-3 lg:mb-4">
   463→            <div className="flex gap-2">
   464→              <ProductAutocomplete
   465→                inputRef={searchInputRef}
   466→                organizationId={profile?.organization_id}
   467→                onSelect={(p, qty) => {
   468→                  if (p.stock_quantity === 0) {
   469→                    toast({
   470→                      variant: "destructive",
   471→                      title: "Rupture de stock",
   472→                      description: `${p.name} n'est pas disponible`,
   473→                    });
   474→                    return;
   475→                  }
   476→                  addToCart(p, qty);
   477→                }}
   478→                placeholder="Rechercher un produit (nom ou code-barres)..."
   479→              />
   480→              <Button
   481→                variant="outline"
   482→                size="icon"
   483→                onClick={() => setIsScannerOpen(true)}
   484→                title="F7 : Scanner un code-barres"
   485→                aria-label="Scanner un code-barres"
   486→              >
   487→                <Camera className="h-4 w-4" />
   488→              </Button>
   489→              <Button
   490→                variant="outline"
   491→                size="icon"
   492→                onClick={() => setShowShortcuts(true)}
   493→                title="Raccourcis clavier"
   494→                aria-label="Raccourcis clavier"
   495→              >
   496→                <Keyboard className="h-4 w-4" />
   497→              </Button>
   498→            </div>
   499→
   500→            <div className="flex items-center justify-between gap-2 flex-wrap">
   501→              <div className="text-xs text-muted-foreground">
   502→                {products.length} produit(s) affiché(s)
   503→                {totalProductCount > 0 && ` / ${totalProductCount} au total`}
   504→              </div>
   505→              <div className="flex items-center gap-2 sm:gap-3">
   506→                <CurrencySelector variant="compact" />
   507→                {/* View mode toggle */}
   508→                <div className="flex items-center border rounded-md">
   509→                  <Button
   510→                    variant={viewMode === "grid" ? "default" : "ghost"}
   511→                    size="icon"
   512→                    className="h-7 w-7 rounded-r-none"
   513→                    onClick={() => setViewMode("grid")}
   514→                    aria-label="Vue grille"
   515→                  >
   516→                    <LayoutGrid className="h-3.5 w-3.5" />
   517→                  </Button>
   518→                  <Button
   519→                    variant={viewMode === "list" ? "default" : "ghost"}
   520→                    size="icon"
   521→                    className="h-7 w-7 rounded-l-none"
   522→                    onClick={() => setViewMode("list")}
   523→                    aria-label="Vue liste"
   524→                  >
   525→                    <List className="h-3.5 w-3.5" />
   526→                  </Button>
   527→                </div>
   528→                <div className="flex items-center gap-1.5 sm:gap-2">
   529→                  <Switch
   530→                    id="show-out-of-stock"
   531→                    checked={showOutOfStock}
   532→                    onCheckedChange={setShowOutOfStock}
   533→                  />
   534→                  <Label htmlFor="show-out-of-stock" className="text-xs cursor-pointer hidden sm:inline">
   535→                    Ruptures
   536→                  </Label>
   537→                </div>
   538→              </div>
   539→            </div>
   540→            
   541→            {/* Category Filters */}
   542→            <div className="flex gap-1.5 sm:gap-2 flex-wrap overflow-x-auto no-scrollbar pb-1">
   543→              <Button
   544→                variant={selectedCategory === null ? "default" : "outline"}
   545→                size="sm"
   546→                onClick={() => setSelectedCategory(null)}
   547→              >
   548→                Tous
   549→              </Button>
   550→              {categories?.map((category) => (
   551→                <Button
   552→                  key={category.id}
   553→                  variant={selectedCategory === category.id ? "default" : "outline"}
   554→                  size="sm"
   555→                  onClick={() => setSelectedCategory(category.id)}
   556→                  style={{
   557→                    backgroundColor:
   558→                      selectedCategory === category.id
   559→                        ? category.color || undefined
   560→                        : undefined,
   561→                  }}
   562→                >
   563→                  <CategoryIcon iconName={category.icon} className="h-3.5 w-3.5" /> <span className="hidden sm:inline">{category.name}</span><span className="sm:hidden">{category.name.length > 8 ? category.name.slice(0, 8) + '…' : category.name}</span>
   564→                </Button>
   565→              ))}
   566→            </div>
   567→          </div>
   568→
   569→          {/* Products Grid */}
   570→          <div className="flex-1 overflow-y-auto">
   571→            {isLoading ? (
   572→              viewMode === "grid" ? (
   573→                <POSProductGridSkeleton />
   574→              ) : (
   575→                <POSProductListSkeleton />
   576→              )
   577→            ) : products.length > 0 ? (
   578→              viewMode === "grid" ? (
   579→                <POSProductGrid
   580→                  products={products}
   581→                  onAddToCart={addToCart}
   582→                  hasMore={hasNextPage ?? false}
   583→                  isLoadingMore={isFetchingNextPage}
   584→                  onLoadMore={fetchNextPage}
   585→                  totalCount={totalProductCount}
   586→                />
   587→              ) : (
   588→                <POSProductList
   589→                  products={products}
   590→                  onAddToCart={addToCart}
   591→                  hasMore={hasNextPage ?? false}
   592→                  isLoadingMore={isFetchingNextPage}
   593→                  onLoadMore={fetchNextPage}
   594→                  totalCount={totalProductCount}
   595→                />
   596→              )
   597→            ) : (
   598→              <div className="text-center py-12">
   599→                <ShoppingCart className="h-12 w-12 mx-auto mb-4 text-muted-foreground opacity-50" />
   600→                <p className="text-muted-foreground">Aucun produit trouvé</p>
   601→              </div>
   602→            )}
   603→          </div>
   604→        </div>
   605→
   606→        {/* Cart Section — hidden on mobile, visible on lg+ */}
   607→        <div className="hidden lg:flex lg:w-80 xl:w-96 flex-shrink-0">
   608→          {isLoading ? (
   609→            <POSCartSkeleton />
   610→          ) : (
   611→            <POSCart
   612→              items={cart}
   613→              total={cartTotal}
   614→              onUpdateQuantity={updateCartQuantity}
   615→              onRemove={removeFromCart}
   616→              onClear={clearCart}
   617→              onCheckout={() => setIsPaymentOpen(true)}
   618→            />
   619→          )}
   620→        </div>
   621→
   622→        {/* Mobile floating cart button */}
   623→        {itemCount > 0 && (
   624→          <div className="lg:hidden fixed bottom-[5.5rem] right-4 z-40">
   625→            <Button
   626→              size="lg"
   627→              className="rounded-full shadow-lg h-14 w-14 relative"
   628→              onClick={() => setIsMobileCartOpen(true)}
   629→              aria-label="Voir le panier"
   630→            >
   631→              <ShoppingCart className="h-6 w-6" />
   632→              <span className="absolute -top-1 -right-1 bg-destructive text-destructive-foreground text-xs font-bold rounded-full h-5 w-5 flex items-center justify-center">
   633→                {itemCount}
   634→              </span>
   635→            </Button>
   636→            <div className="absolute -top-8 left-1/2 -translate-x-1/2 whitespace-nowrap bg-primary text-primary-foreground text-xs font-bold px-2.5 py-1 rounded-md shadow">
   637→              {formatPrice(cartTotal)}
   638→            </div>
   639→          </div>
   640→        )}
   641→
   642→        {/* Mobile Cart Drawer */}
   643→        <MobileCartDrawer
   644→          isOpen={isMobileCartOpen}
   645→          onClose={() => setIsMobileCartOpen(false)}
   646→          items={cart}
   647→          total={cartTotal}
   648→          onUpdateQuantity={updateCartQuantity}
   649→          onRemove={removeFromCart}
   650→          onClear={clearCart}
   651→          onCheckout={() => setIsPaymentOpen(true)}
   652→        />
   653→
   654→        {/* Payment Dialog */}
   655→        <POSPaymentDialog
   656→          isOpen={isPaymentOpen}
   657→          onClose={() => setIsPaymentOpen(false)}
   658→          total={cartTotal}
   659→          onConfirm={(paymentMethod, amountPaid, customerName, customerPhone) =>
   660→            createSaleMutation.mutate({
   661→              paymentMethod,
   662→              amountPaid,
   663→              customerName,
   664→              customerPhone,
   665→            })
   666→          }
   667→          isLoading={createSaleMutation.isPending}
   668→          confirmRef={confirmPaymentRef}
   669→        />
   670→
   671→        {/* Receipt Actions Dialog */}
   672→        <ReceiptActionsDialog
   673→          isOpen={isReceiptOpen}
   674→          onClose={() => setIsReceiptOpen(false)}
   675→          receiptData={lastReceiptData}
   676→        />
   677→
   678→        {/* Barcode Scanner Dialog */}
   679→        <BarcodeScannerDialog
   680→          isOpen={isScannerOpen}
   681→          onClose={() => setIsScannerOpen(false)}
   682→          onScan={handleBarcodeScan}
   683→        />
   684→
   685→        {/* Keyboard Shortcuts Dialog */}
   686→        <Dialog open={showShortcuts} onOpenChange={setShowShortcuts}>
   687→          <DialogContent className="max-w-sm" aria-describedby={undefined}>
   688→            <DialogHeader>
   689→              <DialogTitle className="flex items-center gap-2">
   690→                <Keyboard className="h-5 w-5" />
   691→                Raccourcis clavier
   692→              </DialogTitle>
   693→            </DialogHeader>
   694→            <div className="space-y-2 text-sm">
   695→              {[
   696→                { keys: "F1", action: "Aide raccourcis clavier" },
   697→                { keys: "/ ou Ctrl+K", action: "Rechercher un produit" },
   698→                { keys: "F2", action: "Ouvrir le paiement" },
   699→                { keys: "Ctrl+Entrée", action: "Confirmer le paiement" },
   700→                { keys: "F4", action: "Vider le panier" },
   701→                { keys: "+ / -", action: "Modifier qté dernier article" },
   702→                { keys: "F5", action: "Basculer grille / liste" },
   703→                { keys: "F6", action: "Afficher / masquer ruptures" },
   704→                { keys: "F7", action: "Scanner un code-barres" },
   705→                { keys: "Escape", action: "Fermer / annuler" },
   706→              ].map((shortcut) => (
   707→                <div key={shortcut.keys} className="flex items-center justify-between py-1.5 border-b border-border/50 last:border-0">
   708→                  <span className="text-muted-foreground">{shortcut.action}</span>
   709→                  <kbd className="px-2 py-0.5 bg-muted rounded text-xs font-mono">
   710→                    {shortcut.keys}
   711→                  </kbd>
   712→                </div>
   713→              ))}
   714→            </div>
   715→          </DialogContent>
   716→        </Dialog>
   717→      </div>
   718→    </DashboardLayout>
   719→  );
   720→};
   721→
   722→export default POS;
